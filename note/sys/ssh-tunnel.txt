ssh -L [local_port]:[destination_address]:[destination_port] [username]@[ssh_server]
from mbpro 192.168.1.115: ssh -L 2222:172.25.0.50:22 root@172.25.0.50
from vm bridged: ssh -p 2222 -l root 192.168.1.115
